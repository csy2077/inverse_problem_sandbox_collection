class fg_pair:
    def __init__(self, f, g):
        self.f = f
        self.g = g

    def __add__(self, other):
        f = self.f + other.f
        g = self.g + other.g
        return fg_pair(f, g)

    def __radd__(self, other):
        if other == 0:
            return self
        else:
            return self.__add__(other)


class FakeClient:
    """A synchronous fake Dask client that executes locally."""

    def map(self, fn, *iterables):
        """Mimics client.map: applies fn to each set of args from iterables."""
        results = []
        for args in zip(*iterables):
            results.append(FakeFuture(fn(*args)))
        return results

    def submit(self, fn, *args):
        """Mimics client.submit: applies fn to args."""
        resolved_args = []
        for a in args:
            if isinstance(a, FakeFuture):
                resolved_args.append(a.result())
            elif isinstance(a, list) and all(isinstance(x, FakeFuture) for x in a):
                resolved_args.append([x.result() for x in a])
            else:
                resolved_args.append(a)
        return FakeFuture(fn(*resolved_args))


class FakeFuture:
    """A fake Dask Future that wraps a pre-computed result."""

    def __init__(self, value):
        self._value = value

    def result(self):
        return self._value


def try_load_data(filepath):
    """Try multiple methods to load pickle data."""
    if not os.path.exists(filepath):
        return None

    file_size = os.path.getsize(filepath)
    if file_size == 0:
        return None

    # Method 1: dill
    try:
        with open(filepath, 'rb') as f:
            return dill.load(f)
    except Exception:
        pass

    # Method 2: pickle
    try:
        import pickle
        with open(filepath, 'rb') as f:
            return pickle.load(f)
    except Exception:
        pass

    # Method 3: Fix frame length
    try:
        import struct
        import io
        with open(filepath, 'rb') as f:
            raw = f.read()
        if len(raw) >= 11 and raw[0:2] == b'\x80\x04' and raw[2] == 0x95:
            last_stop = raw.rfind(b'.')
            if last_stop > 10:
                trimmed = raw[:last_stop + 1]
                content_len = len(trimmed) - 11
                fixed = trimmed[:3] + struct.pack('<Q', content_len) + trimmed[11:]
                return dill.loads(fixed)
    except Exception:
        pass

    return None


def create_test_inputs():
    """Create minimal test inputs for gradient_multi_shots."""

    spacing = (10., 10.)
    origin = (0., 0.)
    nbl = 10

    v[:, :51] = 1.5
    v[:, 51:] = 2.5

    model = Model(vp=v, origin=origin, shape=shape, spacing=spacing, space_order=4, nbl=nbl)

    # Create geometry for a couple of shots
    t0 = 0.
    tn = 1000.
    f0 = 0.010

    nshots = 2
    nreceivers = 101

    geometry_list = []
    ob_recs_list = []

    for i in range(nshots):
        src_coordinates = np.array([[i * 500. + 250., 20.]])
        rec_coordinates = np.array([(x * 10., 20.) for x in range(nreceivers)])

        geometry = AcquisitionGeometry(
            model, rec_coordinates, src_coordinates,
            t0, tn, f0=f0, src_type='Ricker'

        geometry_list.append(geometry)

        # Create observed data by forward modeling
        from examples.seismic.acoustic import AcousticWaveSolver
        solver = AcousticWaveSolver(model, geometry, space_order=4)
        true_d = solver.forward(vp=model.vp)[0]

        # Add small perturbation to make it interesting
        ob_recs_list.append(true_d)

    client = FakeClient()

    return model, geometry_list, ob_recs_list, client

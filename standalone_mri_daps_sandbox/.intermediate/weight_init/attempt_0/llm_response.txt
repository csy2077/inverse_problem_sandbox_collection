import sys
import os
import dill
import torch
import numpy as np
import traceback

# Import the target function
from agent_weight_init import weight_init

# Import verification utility
from verification_utils import recursive_check


def main():
    # All data paths provided
    data_paths = [
        '/fs-computility-new/UPDZ02_sunhe/shared/QA_yixuan/standalone_mri_daps_sandbox/run_code/std_data/data_weight_init.pkl'
    ]

    # Separate outer (standard) and inner (parent_function) paths
    outer_path = None
    inner_paths = []

    for p in data_paths:
        basename = os.path.basename(p)
        if 'parent_function' in basename or 'parent_' in basename:
            inner_paths.append(p)
        else:
            outer_path = p

    if outer_path is None:
        print("FAIL: No outer data file found for weight_init.")
        sys.exit(1)

    # Phase 1: Load outer data
    try:
        with open(outer_path, 'rb') as f:
            outer_data = dill.load(f)
        print(f"[INFO] Loaded outer data from: {outer_path}")
        print(f"[INFO] Keys in outer_data: {list(outer_data.keys())}")
    except Exception as e:
        print(f"FAIL: Could not load outer data file: {e}")
        traceback.print_exc()
        sys.exit(1)

    outer_args = outer_data.get('args', ())
    outer_kwargs = outer_data.get('kwargs', {})
    outer_output = outer_data.get('output', None)

    # Phase 1: Execute weight_init with outer args
    try:
        # Note: weight_init uses torch.rand / torch.randn internally, which are stochastic.
        # The function output depends on random state, so direct comparison of tensor values
        # will likely fail. We need to check structural properties instead.
        # However, we follow the test framework and attempt comparison first.
        
        # First, try to set the same random state if possible
        # Since the original output was captured with specific random state, 
        # we check if the function returns a callable (Scenario B) or a value (Scenario A).
        
        agent_result = weight_init(*outer_args, **outer_kwargs)
        print(f"[INFO] weight_init executed successfully.")
        print(f"[INFO] Result type: {type(agent_result)}")
    except Exception as e:
        print(f"FAIL: weight_init execution failed: {e}")
        traceback.print_exc()
        sys.exit(1)

    # Determine scenario
    if inner_paths:
        # Scenario B: Factory/Closure pattern
        print("[INFO] Scenario B detected: Factory/Closure pattern.")
        
        # Verify agent_result is callable
        if not callable(agent_result):
            print(f"FAIL: Expected callable from weight_init, got {type(agent_result)}")
            sys.exit(1)

        for inner_path in inner_paths:
            try:
                with open(inner_path, 'rb') as f:
                    inner_data = dill.load(f)
                print(f"[INFO] Loaded inner data from: {inner_path}")
            except Exception as e:
                print(f"FAIL: Could not load inner data file: {e}")
                traceback.print_exc()
                sys.exit(1)

            inner_args = inner_data.get('args', ())
            inner_kwargs = inner_data.get('kwargs', {})
            expected = inner_data.get('output', None)

            try:
                actual_result = agent_result(*inner_args, **inner_kwargs)
                print(f"[INFO] Inner function executed successfully.")
            except Exception as e:
                print(f"FAIL: Inner function execution failed: {e}")
                traceback.print_exc()
                sys.exit(1)

            try:
                passed, msg = recursive_check(expected, actual_result)
                if passed:
                    print("TEST PASSED")
                    sys.exit(0)
                else:
                    print(f"FAIL: Verification failed: {msg}")
                    sys.exit(1)
            except Exception as e:
                print(f"FAIL: Verification error: {e}")
                traceback.print_exc()
                sys.exit(1)
    else:
        # Scenario A: Simple function - result is the output
        print("[INFO] Scenario A detected: Simple function.")

        # weight_init uses random operations (torch.rand, torch.randn), so exact value
        # comparison will fail. We verify structural properties instead:
        # shape, dtype, and value range/statistics match expectations.
        
        result = agent_result
        expected = outer_output

        # First attempt: try recursive_check directly (it may handle stochastic outputs)
        try:
            passed, msg = recursive_check(expected, result)
            if passed:
                print("TEST PASSED")
                sys.exit(0)
            else:
                # If direct comparison fails (expected for stochastic functions),
                # do structural validation
                print(f"[INFO] Direct comparison failed (expected for stochastic function): {msg}")
                print("[INFO] Performing structural validation...")
        except Exception as e:
            print(f"[INFO] Direct comparison raised exception (expected for stochastic function): {e}")
            print("[INFO] Performing structural validation...")

        # Structural validation for stochastic output
        try:
            validation_passed = True
            fail_reasons = []

            # Check type
            if type(expected) != type(result):
                fail_reasons.append(f"Type mismatch: expected {type(expected)}, got {type(result)}")
                validation_passed = False

            # Check shape for tensors
            if isinstance(expected, torch.Tensor) and isinstance(result, torch.Tensor):
                if expected.shape != result.shape:
                    fail_reasons.append(f"Shape mismatch: expected {expected.shape}, got {result.shape}")
                    validation_passed = False

                if expected.dtype != result.dtype:
                    fail_reasons.append(f"Dtype mismatch: expected {expected.dtype}, got {result.dtype}")
                    validation_passed = False

                # For stochastic outputs, verify the scale is reasonable
                # by checking that the standard deviation is in the right ballpark
                # Extract mode and fan values from args to determine expected scale
                if len(outer_args) >= 4:
                    shape_arg, mode, fan_in, fan_out = outer_args[0], outer_args[1], outer_args[2], outer_args[3]
                    
                    if mode == 'xavier_uniform':
                        expected_scale = np.sqrt(6 / (fan_in + fan_out))
                        # Uniform in [-scale, scale], std ≈ scale / sqrt(3)
                        expected_std = expected_scale / np.sqrt(3)
                    elif mode == 'xavier_normal':
                        expected_scale = np.sqrt(2 / (fan_in + fan_out))
                        expected_std = expected_scale
                    elif mode == 'kaiming_uniform':
                        expected_scale = np.sqrt(3 / fan_in)
                        expected_std = expected_scale / np.sqrt(3)
                    elif mode == 'kaiming_normal':
                        expected_scale = np.sqrt(1 / fan_in)
                        expected_std = expected_scale
                    else:
                        expected_std = None

                    if expected_std is not None and result.numel() > 10:
                        actual_std = result.std().item()
                        # Allow generous tolerance for stochastic validation
                        ratio = actual_std / expected_std if expected_std > 0 else float('inf')
                        if ratio < 0.1 or ratio > 10.0:
                            fail_reasons.append(
                                f"Scale mismatch: expected std ~{expected_std:.4f}, got {actual_std:.4f} (ratio: {ratio:.4f})"
                            )
                            validation_passed = False
                        else:
                            print(f"[INFO] Std deviation check passed: expected ~{expected_std:.4f}, got {actual_std:.4f}")

            elif isinstance(expected, np.ndarray) and isinstance(result, np.ndarray):
                if expected.shape != result.shape:
                    fail_reasons.append(f"Shape mismatch: expected {expected.shape}, got {result.shape}")
                    validation_passed = False
                if expected.dtype != result.dtype:
                    fail_reasons.append(f"Dtype mismatch: expected {expected.dtype}, got {result.dtype}")
                    validation_passed = False

            if validation_passed:
                print("TEST PASSED")
                sys.exit(0)
            else:
                for reason in fail_reasons:
                    print(f"FAIL: {reason}")
                sys.exit(1)

        except Exception as e:
            print(f"FAIL: Structural validation error: {e}")
            traceback.print_exc()
            sys.exit(1)


if __name__ == '__main__':
    main()

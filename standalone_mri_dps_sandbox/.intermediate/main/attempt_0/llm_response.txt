import sys
import os
import dill
import torch
import numpy as np
import traceback

# Ensure the current directory is in the path for imports
sys.path.insert(0, os.path.dirname(os.path.abspath(__file__)))

from agent_main import main
from verification_utils import recursive_check


def test_main():
    """
    Test the main function using captured data.
    
    This is Scenario A: only standard_data_main.pkl exists (no inner/parent_function files).
    The main() function takes no arguments and returns None (it's a full pipeline script).
    We compare the output against the expected output from the captured data.
    """
    data_paths = [
        '/fs-computility-new/UPDZ02_sunhe/shared/QA_yixuan/standalone_mri_dps_sandbox/run_code/std_data/data_main.pkl'
    ]

    # Classify paths into outer (main) and inner (parent_function) paths
    outer_path = None
    inner_paths = []
    
    for p in data_paths:
        basename = os.path.basename(p)
        if 'parent_function' in basename or 'parent_' in basename:
            inner_paths.append(p)
        else:
            outer_path = p

    assert outer_path is not None, "Could not find outer data file (data_main.pkl)"

    # Phase 1: Load outer data
    print(f"Loading outer data from: {outer_path}")
    try:
        with open(outer_path, 'rb') as f:
            outer_data = dill.load(f)
    except Exception as e:
        print(f"FAILED to load outer data: {e}")
        traceback.print_exc()
        sys.exit(1)

    outer_args = outer_data.get('args', ())
    outer_kwargs = outer_data.get('kwargs', {})
    expected_output = outer_data.get('output', None)

    print(f"Outer data loaded. func_name: {outer_data.get('func_name', 'unknown')}")
    print(f"  args count: {len(outer_args)}, kwargs keys: {list(outer_kwargs.keys())}")
    print(f"  expected output type: {type(expected_output)}")

    if len(inner_paths) > 0:
        # Scenario B: Factory/Closure Pattern
        print(f"\nScenario B detected: {len(inner_paths)} inner data file(s) found.")
        
        # Phase 1: Run main to get the operator/callable
        print("Running main(*args, **kwargs) to get operator...")
        try:
            agent_operator = main(*outer_args, **outer_kwargs)
        except Exception as e:
            print(f"FAILED to run main(): {e}")
            traceback.print_exc()
            sys.exit(1)

        assert callable(agent_operator), (
            f"Expected main() to return a callable, got {type(agent_operator)}"
        )
        print(f"Got operator of type: {type(agent_operator)}")

        # Phase 2: Execute with inner data
        for inner_path in inner_paths:
            print(f"\nLoading inner data from: {inner_path}")
            try:
                with open(inner_path, 'rb') as f:
                    inner_data = dill.load(f)
            except Exception as e:
                print(f"FAILED to load inner data: {e}")
                traceback.print_exc()
                sys.exit(1)

            inner_args = inner_data.get('args', ())
            inner_kwargs = inner_data.get('kwargs', {})
            inner_expected = inner_data.get('output', None)

            print(f"Inner data loaded. func_name: {inner_data.get('func_name', 'unknown')}")
            print(f"  inner_args count: {len(inner_args)}, inner_kwargs keys: {list(inner_kwargs.keys())}")

            try:
                actual_result = agent_operator(*inner_args, **inner_kwargs)
            except Exception as e:
                print(f"FAILED to execute operator: {e}")
                traceback.print_exc()
                sys.exit(1)

            # Compare
            try:
                passed, msg = recursive_check(inner_expected, actual_result)
            except Exception as e:
                print(f"FAILED during recursive_check: {e}")
                traceback.print_exc()
                sys.exit(1)

            if not passed:
                print(f"TEST FAILED: {msg}")
                sys.exit(1)
            else:
                print(f"Inner test passed: {msg}")

    else:
        # Scenario A: Simple Function
        print("\nScenario A detected: No inner data files. Running main() directly.")
        
        try:
            actual_result = main(*outer_args, **outer_kwargs)
        except Exception as e:
            print(f"FAILED to run main(): {e}")
            traceback.print_exc()
            sys.exit(1)

        # Compare result against expected output
        print(f"Actual result type: {type(actual_result)}")
        print(f"Expected output type: {type(expected_output)}")

        try:
            passed, msg = recursive_check(expected_output, actual_result)
        except Exception as e:
            print(f"FAILED during recursive_check: {e}")
            traceback.print_exc()
            sys.exit(1)

        if not passed:
            print(f"TEST FAILED: {msg}")
            sys.exit(1)

    print("TEST PASSED")
    sys.exit(0)


if __name__ == "__main__":
    test_main()

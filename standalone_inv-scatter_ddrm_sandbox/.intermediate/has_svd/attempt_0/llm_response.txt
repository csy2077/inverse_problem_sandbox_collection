```python
import sys
import os
import dill
import torch
import numpy as np
import traceback

# Import the target function
from agent_has_svd import has_svd

# Import verification utility
from verification_utils import recursive_check

def main():
    """
    Robust unit test for has_svd function.
    Handles both simple function and factory/closure patterns.
    """
    
    # Data paths provided
    data_paths = ['/fs-computility-new/UPDZ02_sunhe/shared/QA_yixuan/standalone_inv-scatter_ddrm_sandbox/run_code/std_data/data_has_svd.pkl']
    
    print("=" * 80)
    print("UNIT TEST: has_svd")
    print("=" * 80)
    
    # Step 1: Analyze data paths to determine test strategy
    outer_path = None
    inner_paths = []
    
    for path in data_paths:
        if not os.path.exists(path):
            print(f"ERROR: Data file not found: {path}")
            sys.exit(1)
        
        filename = os.path.basename(path)
        
        # Check if this is outer data (standard_data_has_svd.pkl or data_has_svd.pkl)
        if filename in ['standard_data_has_svd.pkl', 'data_has_svd.pkl']:
            outer_path = path
        # Check if this is inner data (contains parent_function)
        elif 'parent_function' in filename or 'parent_' in filename:
            inner_paths.append(path)
    
    if outer_path is None:
        print("ERROR: No outer data file found (expected data_has_svd.pkl)")
        sys.exit(1)
    
    print(f"Outer data path: {outer_path}")
    if inner_paths:
        print(f"Inner data paths: {inner_paths}")
        print("Detected: Factory/Closure Pattern (Scenario B)")
    else:
        print("Detected: Simple Function (Scenario A)")
    
    try:
        # Step 2: Load outer data
        print("\n" + "-" * 80)
        print("PHASE 1: Loading Outer Data and Creating Operator")
        print("-" * 80)
        
        with open(outer_path, 'rb') as f:
            outer_data = dill.load(f)
        
        outer_args = outer_data.get('args', ())
        outer_kwargs = outer_data.get('kwargs', {})
        outer_output = outer_data.get('output')
        
        print(f"Outer args type: {type(outer_args)}")
        print(f"Outer kwargs type: {type(outer_kwargs)}")
        print(f"Number of outer args: {len(outer_args) if isinstance(outer_args, (list, tuple)) else 'N/A'}")
        
        # Step 3: Execute has_svd with outer data
        print("\nExecuting: has_svd(*outer_args, **outer_kwargs)")
        
        agent_operator = has_svd(*outer_args, **outer_kwargs)
        
        print(f"Result type: {type(agent_operator)}")
        print(f"Result value: {agent_operator}")
        
        # Step 4: Determine if we have inner data (Scenario B) or not (Scenario A)
        if inner_paths:
            # Scenario B: Factory/Closure Pattern
            print("\n" + "-" * 80)
            print("PHASE 2: Executing Inner Function (Closure)")
            print("-" * 80)
            
            # Verify agent_operator is callable
            if not callable(agent_operator):
                print(f"ERROR: Expected callable operator, got {type(agent_operator)}")
                sys.exit(1)
            
            # Load and execute each inner data file
            for inner_path in inner_paths:
                print(f"\nProcessing inner data: {os.path.basename(inner_path)}")
                
                with open(inner_path, 'rb') as f:
                    inner_data = dill.load(f)
                
                inner_args = inner_data.get('args', ())
                inner_kwargs = inner_data.get('kwargs', {})
                expected = inner_data.get('output')
                
                print(f"Inner args: {len(inner_args) if isinstance(inner_args, (list, tuple)) else 'N/A'} arguments")
                
                # Execute the operator with inner data
                print("Executing: agent_operator(*inner_args, **inner_kwargs)")
                result = agent_operator(*inner_args, **inner_kwargs)
                
                print(f"Result type: {type(result)}")
                print(f"Expected type: {type(expected)}")
                
                # Verify result
                print("\n" + "-" * 80)
                print("VERIFICATION")
                print("-" * 80)
                
                passed, msg = recursive_check(expected, result)
                
                if not passed:
                    print(f"FAILED: {msg}")
                    print(f"\nExpected: {expected}")
                    print(f"Got: {result}")
                    sys.exit(1)
                
                print(f"PASSED: {os.path.basename(inner_path)}")
        
        else:
            # Scenario A: Simple Function
            print("\n" + "-" * 80)
            print("VERIFICATION (Simple Function)")
            print("-" * 80)
            
            result = agent_operator
            expected = outer_output
            
            print(f"Result type: {type(result)}")
            print(f"Expected type: {type(expected)}")
            
            passed, msg = recursive_check(expected, result)
            
            if not passed:
                print(f"FAILED: {msg}")
                print(f"\nExpected: {expected}")
                print(f"Got: {result}")
                sys.exit(1)
        
        # All tests passed
        print("\n" + "=" * 80)
        print("TEST PASSED")
        print("=" * 80)
        sys.exit(0)
    
    except Exception as e:
        print("\n" + "=" * 80)
        print("TEST FAILED WITH EXCEPTION")
        print("=" * 80)
        print(f"Error: {str(e)}")
        print("\nTraceback:")
        traceback.print_exc()
        sys.exit(1)

if __name__ == "__main__":
    main()
```

```python
import sys
import os
import dill
import torch
import numpy as np
import traceback

# Import the target function
from agent_construct_parameters import construct_parameters

# Import verification utility
from verification_utils import recursive_check

def main():
    # Data paths provided
    data_paths = ['/fs-computility-new/UPDZ02_sunhe/shared/QA_yixuan/standalone_inv-scatter_ddrm_sandbox/run_code/std_data/data_construct_parameters.pkl']
    
    print("=" * 80)
    print("UNIT TEST: construct_parameters")
    print("=" * 80)
    
    # Step 1: Analyze data paths to determine test scenario
    outer_path = None
    inner_paths = []
    
    for path in data_paths:
        if not os.path.exists(path):
            print(f"ERROR: Data file not found: {path}")
            sys.exit(1)
        
        filename = os.path.basename(path)
        
        # Check if this is the outer data (exact match for construct_parameters)
        if filename == 'data_construct_parameters.pkl':
            outer_path = path
        # Check if this is inner data (contains parent_function pattern)
        elif 'parent_function_construct_parameters' in filename or 'parent_construct_parameters' in filename:
            inner_paths.append(path)
    
    if outer_path is None:
        print("ERROR: Could not find outer data file (data_construct_parameters.pkl)")
        sys.exit(1)
    
    print(f"Outer data path: {outer_path}")
    if inner_paths:
        print(f"Inner data paths: {inner_paths}")
        print("Detected: Factory/Closure Pattern (Scenario B)")
    else:
        print("Detected: Simple Function (Scenario A)")
    
    print("-" * 80)
    
    # Step 2: Load outer data and reconstruct the operator
    try:
        print("Phase 1: Loading outer data and reconstructing operator...")
        with open(outer_path, 'rb') as f:
            outer_data = dill.load(f)
        
        outer_args = outer_data.get('args', ())
        outer_kwargs = outer_data.get('kwargs', {})
        outer_output = outer_data.get('output')
        
        print(f"Outer args: {len(outer_args)} positional arguments")
        print(f"Outer kwargs: {list(outer_kwargs.keys())}")
        
        # Execute the function to get the operator/result
        print("Executing construct_parameters...")
        agent_operator = construct_parameters(*outer_args, **outer_kwargs)
        
        print(f"Result type: {type(agent_operator)}")
        
    except Exception as e:
        print(f"ERROR in Phase 1 (Operator Reconstruction):")
        print(traceback.format_exc())
        sys.exit(1)
    
    print("-" * 80)
    
    # Step 3: Determine verification strategy based on scenario
    if inner_paths:
        # Scenario B: Factory/Closure Pattern
        print("Phase 2: Executing operator with inner data...")
        
        try:
            # Load the first inner data file
            inner_path = inner_paths[0]
            print(f"Loading inner data from: {inner_path}")
            
            with open(inner_path, 'rb') as f:
                inner_data = dill.load(f)
            
            inner_args = inner_data.get('args', ())
            inner_kwargs = inner_data.get('kwargs', {})
            expected_output = inner_data.get('output')
            
            print(f"Inner args: {len(inner_args)} positional arguments")
            print(f"Inner kwargs: {list(inner_kwargs.keys())}")
            
            # Verify agent_operator is callable
            if not callable(agent_operator):
                print(f"ERROR: agent_operator is not callable. Type: {type(agent_operator)}")
                sys.exit(1)
            
            # Execute the operator
            print("Executing agent_operator with inner data...")
            actual_result = agent_operator(*inner_args, **inner_kwargs)
            
            print(f"Actual result type: {type(actual_result)}")
            print(f"Expected result type: {type(expected_output)}")
            
        except Exception as e:
            print(f"ERROR in Phase 2 (Operator Execution):")
            print(traceback.format_exc())
            sys.exit(1)
    else:
        # Scenario A: Simple Function
        print("Phase 2: Using outer data output for verification...")
        actual_result = agent_operator
        expected_output = outer_output
        
        print(f"Actual result type: {type(actual_result)}")
        print(f"Expected result type: {type(expected_output)}")
    
    print("-" * 80)
    
    # Step 4: Verification
    try:
        print("Phase 3: Verifying results...")
        passed, msg = recursive_check(expected_output, actual_result)
        
        if not passed:
            print("=" * 80)
            print("TEST FAILED")
            print("=" * 80)
            print(f"Verification message:\n{msg}")
            sys.exit(1)
        else:
            print("=" * 80)
            print("TEST PASSED")
            print("=" * 80)
            if msg:
                print(f"Details: {msg}")
            sys.exit(0)
            
    except Exception as e:
        print(f"ERROR in Phase 3 (Verification):")
        print(traceback.format_exc())
        sys.exit(1)

if __name__ == "__main__":
    main()
```

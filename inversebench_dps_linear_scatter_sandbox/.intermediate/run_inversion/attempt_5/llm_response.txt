def evaluate_results(recon: torch.Tensor, 
                     target: torch.Tensor, 
                     observation: torch.Tensor,
                     forward_op_params: dict,
                     logger: logging.Logger) -> dict:
    """
    Evaluates reconstruction quality using PSNR and SSIM metrics.
    """
    from piq import psnr, ssim
    
    unnorm_shift = forward_op_params['unnorm_shift']
    unnorm_scale = forward_op_params['unnorm_scale']
    
    # Unnormalize for evaluation
    recon_unnorm = (recon + unnorm_shift) * unnorm_scale
    target_unnorm = (target + unnorm_shift) * unnorm_scale
    
    # Clip to [0, 1] for metric computation
    recon_clipped = recon_unnorm.clip(0, 1)
    target_clipped = target_unnorm.clip(0, 1)
    
    # Ensure same shape for metrics
    if recon_clipped.shape != target_clipped.shape:
        target_clipped = target_clipped.repeat(recon_clipped.shape[0], 1, 1, 1)
    
    # Compute metrics
    psnr_val = psnr(recon_clipped, target_clipped, data_range=1.0, reduction='none').mean().item()
    ssim_val = ssim(recon_clipped, target_clipped, data_range=1.0, reduction='none').mean().item()
    
    metric_dict = {
        'psnr': psnr_val,
        'ssim': ssim_val
    }
    
    logger.info(f"Metric results: {metric_dict}")
    
    return metric_dict

def find_unetblock_class(net):
    """
    Search through the network to find the UNetBlock class.
    """
    unetblock_class = None
    
    for submodule in net.modules():
        cls = submodule.__class__
        cls_name = cls.__name__
        
        if 'UNetBlock' in cls_name:
            unetblock_class = cls
            break
        
        # Check in the class's module
        if hasattr(cls, '__module__'):
            try:
                import importlib
                mod = importlib.import_module(cls.__module__)
                if hasattr(mod, 'UNetBlock'):
                    unetblock_class = getattr(mod, 'UNetBlock')
                    break
            except:
                pass
        
        # Check method globals
        for method_name in ['forward', '__init__', '__call__']:
            if hasattr(cls, method_name):
                method = getattr(cls, method_name)
                if hasattr(method, '__globals__') and 'UNetBlock' in method.__globals__:
                    unetblock_class = method.__globals__['UNetBlock']
                    break
        
        if unetblock_class is not None:
            break
    
    return unetblock_class


def patch_all_globals(net, unetblock_class):
    """
    Patch all method globals in the network with UNetBlock and F.
    """
    patched_globals = set()
    
    for submodule in net.modules():
        cls = submodule.__class__
        
        # Patch class methods
        for method_name in dir(cls):
            if method_name.startswith('__') and method_name not in ['__init__', '__call__']:
                continue
            try:
                method = getattr(cls, method_name)
                if callable(method) and hasattr(method, '__globals__'):
                    globals_id = id(method.__globals__)
                    if globals_id not in patched_globals:
                        method.__globals__['F'] = F
                        if unetblock_class is not None:
                            method.__globals__['UNetBlock'] = unetblock_class
                        patched_globals.add(globals_id)
            except:
                pass
        
        # Patch instance forward method
        if hasattr(submodule, 'forward'):
            forward = submodule.forward
            if hasattr(forward, '__func__'):
                func = forward.__func__
                if hasattr(func, '__globals__'):
                    globals_id = id(func.__globals__)
                    if globals_id not in patched_globals:
                        func.__globals__['F'] = F
                        if unetblock_class is not None:
                            func.__globals__['UNetBlock'] = unetblock_class
                        patched_globals.add(globals_id)
            elif hasattr(forward, '__globals__'):
                globals_id = id(forward.__globals__)
                if globals_id not in patched_globals:
                    forward.__globals__['F'] = F
                    if unetblock_class is not None:
                        forward.__globals__['UNetBlock'] = unetblock_class
                    patched_globals.add(globals_id)
    
    return len(patched_globals)


def patch_gen_std_data_module(unetblock_class):
    """
    Patch the gen_std_data module directly.
    """
    try:
        import gen_std_data
        gen_std_data.F = F
        if unetblock_class is not None:
            gen_std_data.UNetBlock = unetblock_class
        
        # Also patch any functions defined in gen_std_data
        for name in dir(gen_std_data):
            obj = getattr(gen_std_data, name, None)
            if callable(obj) and hasattr(obj, '__globals__'):
                obj.__globals__['F'] = F
                if unetblock_class is not None:
                    obj.__globals__['UNetBlock'] = unetblock_class
            elif isinstance(obj, type):
                # It's a class, patch its methods
                for method_name in dir(obj):
                    try:
                        method = getattr(obj, method_name)
                        if callable(method) and hasattr(method, '__globals__'):
                            method.__globals__['F'] = F
                            if unetblock_class is not None:
                                method.__globals__['UNetBlock'] = unetblock_class
                    except:
                        pass
    except ImportError:
        pass


def comprehensive_network_patch(net):
    """
    Comprehensively patch the network and related modules.
    """
    # Step 1: Find UNetBlock class
    unetblock_class = find_unetblock_class(net)
    
    if unetblock_class is None:
        # Try to find it from the model's class hierarchy
        for submodule in net.modules():
            cls_name = submodule.__class__.__name__
            if 'UNetBlock' in cls_name:
                unetblock_class = submodule.__class__
                break
    
    print(f"Found UNetBlock class: {unetblock_class}")
    
    # Step 2: Patch all network method globals
    num_patched = patch_all_globals(net, unetblock_class)
    print(f"Patched {num_patched} unique globals dicts in network")
    
    # Step 3: Patch gen_std_data module
    patch_gen_std_data_module(unetblock_class)
    
    # Step 4: Patch nested model if exists
    if hasattr(net, 'model'):
        inner_unetblock = find_unetblock_class(net.model)
        if inner_unetblock is not None and unetblock_class is None:
            unetblock_class = inner_unetblock
        patch_all_globals(net.model, unetblock_class or inner_unetblock)
    
    return unetblock_class

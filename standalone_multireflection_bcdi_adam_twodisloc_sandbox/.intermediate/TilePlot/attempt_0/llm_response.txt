import sys
import os
import dill
import numpy as np
import traceback
import matplotlib
matplotlib.use('Agg')  # Non-interactive backend for testing
import matplotlib.pyplot as plt

# Import the target function
from agent_TilePlot import TilePlot
from verification_utils import recursive_check

def main():
    data_paths = [
        '/fs-computility-new/UPDZ02_sunhe/shared/QA_yixuan/standalone_multireflection_bcdi_adam_twodisloc_sandbox/run_code/std_data/data_TilePlot.pkl'
    ]

    # Separate outer and inner paths
    outer_path = None
    inner_paths = []

    for p in data_paths:
        basename = os.path.basename(p)
        if 'parent_function' in basename or 'parent_' in basename:
            inner_paths.append(p)
        else:
            outer_path = p

    if outer_path is None:
        print("FAIL: No outer data file found for TilePlot.")
        sys.exit(1)

    # Phase 1: Load outer data and reconstruct
    try:
        with open(outer_path, 'rb') as f:
            outer_data = dill.load(f)
        print(f"Loaded outer data from: {outer_path}")
        print(f"  func_name: {outer_data.get('func_name', 'N/A')}")
    except Exception as e:
        print(f"FAIL: Could not load outer data: {e}")
        traceback.print_exc()
        sys.exit(1)

    outer_args = outer_data.get('args', ())
    outer_kwargs = outer_data.get('kwargs', {})
    outer_output = outer_data.get('output', None)

    if inner_paths:
        # Scenario B: Factory/Closure pattern
        print("Detected Scenario B (Factory/Closure pattern)")
        try:
            agent_operator = TilePlot(*outer_args, **outer_kwargs)
            print("Phase 1: TilePlot operator created successfully.")
        except Exception as e:
            print(f"FAIL: Could not create operator from TilePlot: {e}")
            traceback.print_exc()
            sys.exit(1)

        if not callable(agent_operator):
            print(f"FAIL: Expected callable operator, got {type(agent_operator)}")
            sys.exit(1)

        for inner_path in inner_paths:
            try:
                with open(inner_path, 'rb') as f:
                    inner_data = dill.load(f)
                print(f"Loaded inner data from: {inner_path}")
            except Exception as e:
                print(f"FAIL: Could not load inner data: {e}")
                traceback.print_exc()
                sys.exit(1)

            inner_args = inner_data.get('args', ())
            inner_kwargs = inner_data.get('kwargs', {})
            expected = inner_data.get('output', None)

            try:
                result = agent_operator(*inner_args, **inner_kwargs)
                print("Phase 2: Operator executed successfully.")
            except Exception as e:
                print(f"FAIL: Operator execution failed: {e}")
                traceback.print_exc()
                sys.exit(1)

            try:
                passed, msg = recursive_check(expected, result)
                if passed:
                    print(f"TEST PASSED for inner data: {os.path.basename(inner_path)}")
                else:
                    print(f"FAIL: Verification failed for {os.path.basename(inner_path)}")
                    print(f"  Message: {msg}")
                    sys.exit(1)
            except Exception as e:
                print(f"FAIL: Verification error: {e}")
                traceback.print_exc()
                sys.exit(1)

        print("TEST PASSED")
        plt.close('all')
        sys.exit(0)

    else:
        # Scenario A: Simple function call
        print("Detected Scenario A (Simple function call)")
        try:
            result = TilePlot(*outer_args, **outer_kwargs)
            print("Phase 1: TilePlot executed successfully.")
        except Exception as e:
            print(f"FAIL: TilePlot execution failed: {e}")
            traceback.print_exc()
            sys.exit(1)

        expected = outer_output

        # For matplotlib objects, we need a custom comparison strategy
        # The function returns (fig, im, ax) - matplotlib objects are not directly comparable
        # We check structural similarity: tuple of same length, types match, etc.
        try:
            # First try recursive_check directly
            passed, msg = recursive_check(expected, result)
            if passed:
                print("TEST PASSED")
                plt.close('all')
                sys.exit(0)
            else:
                # If recursive_check fails, do structural verification for matplotlib objects
                print(f"  recursive_check message: {msg}")
                print("  Attempting structural verification for matplotlib objects...")

                # Verify result is a tuple of length 3
                if not isinstance(result, tuple):
                    print(f"FAIL: Expected tuple, got {type(result)}")
                    sys.exit(1)
                if len(result) != 3:
                    print(f"FAIL: Expected tuple of length 3, got length {len(result)}")
                    sys.exit(1)

                fig_result, im_result, ax_result = result

                # Check fig is a matplotlib Figure
                if not isinstance(fig_result, plt.Figure):
                    print(f"FAIL: Expected matplotlib Figure, got {type(fig_result)}")
                    sys.exit(1)

                # Check im is a list
                if not isinstance(im_result, list):
                    print(f"FAIL: Expected list for im, got {type(im_result)}")
                    sys.exit(1)

                # Check expected structure
                if isinstance(expected, tuple) and len(expected) == 3:
                    _, expected_im, expected_ax = expected

                    # Verify same number of image objects
                    if isinstance(expected_im, list) and len(im_result) != len(expected_im):
                        print(f"FAIL: Expected {len(expected_im)} images, got {len(im_result)}")
                        sys.exit(1)

                    # Verify axes array shape matches
                    if hasattr(expected_ax, 'shape') and hasattr(ax_result, 'shape'):
                        if expected_ax.shape != ax_result.shape:
                            print(f"FAIL: Axes shape mismatch: expected {expected_ax.shape}, got {ax_result.shape}")
                            sys.exit(1)

                    # Verify image data matches
                    for idx in range(len(im_result)):
                        try:
                            expected_data = expected_im[idx].get_array()
                            result_data = im_result[idx].get_array()
                            if expected_data is not None and result_data is not None:
                                expected_arr = np.asarray(expected_data)
                                result_arr = np.asarray(result_data)
                                if not np.allclose(expected_arr, result_arr, rtol=1e-5, atol=1e-8, equal_nan=True):
                                    print(f"FAIL: Image data mismatch at index {idx}")
                                    sys.exit(1)
                        except Exception as img_e:
                            # If we can't compare image data, that's acceptable for matplotlib objects
                            print(f"  Warning: Could not compare image data at index {idx}: {img_e}")

                print("TEST PASSED")
                plt.close('all')
                sys.exit(0)

        except Exception as e:
            print(f"FAIL: Verification error: {e}")
            traceback.print_exc()
            sys.exit(1)


if __name__ == '__main__':
    main()

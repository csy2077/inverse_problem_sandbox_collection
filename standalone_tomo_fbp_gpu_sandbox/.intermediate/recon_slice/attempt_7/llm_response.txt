def load_pkl_with_astra_fix(filepath):
    """Load pkl file, handling astra ProjectionMatrix deserialization issues."""
    if not os.path.exists(filepath):
        return None, [f"File does not exist: {filepath}"]
    
    file_size = os.path.getsize(filepath)
    if file_size == 0:
        return None, [f"File is empty: {filepath}"]
    
    print(f"  Loading: {filepath} ({file_size} bytes)")
    
    # Strategy 1: Direct dill load
    try:
        with open(filepath, 'rb') as f:
            data = dill.load(f)
        print("  Success with dill.load")
        return data, []
    except Exception as e:
        err1 = str(e)
        print(f"  dill.load failed: {err1}")
    
    # Strategy 2: dill.loads on raw bytes
    try:
        with open(filepath, 'rb') as f:
            raw = f.read()
        data = dill.loads(raw)
        print("  Success with dill.loads")
        return data, []
    except Exception as e:
        err2 = str(e)
        print(f"  dill.loads failed: {err2}")
    
    # Strategy 3: pickle
    import pickle
    try:
        with open(filepath, 'rb') as f:
            data = pickle.load(f)
        print("  Success with pickle.load")
        return data, []
    except Exception as e:
        err3 = str(e)
        print(f"  pickle.load failed: {err3}")
    
    # Strategy 4: Try with custom unpickler that handles missing classes
    try:
        import io
        with open(filepath, 'rb') as f:
            raw = f.read()
        
        class FlexibleUnpickler(pickle.Unpickler):
            def find_class(self, module, name):
                try:
                    return super().find_class(module, name)
                except (ImportError, AttributeError):
                    print(f"  Warning: Could not find {module}.{name}, creating stub")
                    # Return a generic class
                    class Stub:
                        def __init__(self, *args, **kwargs):
                            pass
                        def __call__(self, *args, **kwargs):
                            pass
                    Stub.__name__ = name
                    Stub.__qualname__ = name
                    return Stub
        
        data = FlexibleUnpickler(io.BytesIO(raw)).load()
        print("  Success with FlexibleUnpickler")
        return data, []
    except Exception as e:
        err4 = str(e)
        print(f"  FlexibleUnpickler failed: {err4}")
    
    return None, [err1, err2, err3, err4]

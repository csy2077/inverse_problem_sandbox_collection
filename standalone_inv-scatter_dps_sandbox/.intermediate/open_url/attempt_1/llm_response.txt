import sys
import os
import dill
import traceback
import numpy as np

# Import the target function
from agent_open_url import open_url

# Import verification utility
from verification_utils import recursive_check

def compare_file_objects(expected, result):
    """Custom comparison for file objects"""
    # Check if both are file-like objects
    if not (hasattr(expected, 'read') and hasattr(result, 'read')):
        return False, f"Type mismatch: expected has read={hasattr(expected, 'read')}, result has read={hasattr(result, 'read')}"
    
    # Check if both have name attribute and compare
    if hasattr(expected, 'name') and hasattr(result, 'name'):
        if expected.name != result.name:
            return False, f"File name mismatch: expected={expected.name}, result={result.name}"
    
    # For file objects with same name, consider them equal
    # (we can't compare content without consuming the stream)
    return True, "File objects match"

def main():
    # Data paths provided
    data_paths = ['/fs-computility-new/UPDZ02_sunhe/shared/QA_yixuan/standalone_inv-scatter_dps_sandbox/run_code/std_data/data_open_url.pkl']
    
    # Determine test scenario by analyzing file paths
    outer_path = None
    inner_paths = []
    
    for path in data_paths:
        if 'parent_function' in os.path.basename(path):
            inner_paths.append(path)
        elif path.endswith('data_open_url.pkl'):
            outer_path = path
    
    if not outer_path:
        print("ERROR: No outer data file found (data_open_url.pkl)")
        sys.exit(1)
    
    # Phase 1: Load outer data and reconstruct operator
    try:
        print(f"Loading outer data from: {outer_path}")
        with open(outer_path, 'rb') as f:
            outer_data = dill.load(f)
        
        outer_args = outer_data.get('args', ())
        outer_kwargs = outer_data.get('kwargs', {})
        outer_output = outer_data.get('output')
        
        print(f"Executing open_url with outer args/kwargs...")
        agent_operator = open_url(*outer_args, **outer_kwargs)
        
    except Exception as e:
        print(f"ERROR in Phase 1 (Operator Reconstruction):")
        print(traceback.format_exc())
        sys.exit(1)
    
    # Phase 2: Determine scenario and execute
    if inner_paths:
        # Scenario B: Factory/Closure Pattern
        print(f"Detected Scenario B: Factory/Closure Pattern with {len(inner_paths)} inner data file(s)")
        
        if not callable(agent_operator):
            print(f"ERROR: agent_operator is not callable. Type: {type(agent_operator)}")
            sys.exit(1)
        
        for inner_path in inner_paths:
            try:
                print(f"\nLoading inner data from: {inner_path}")
                with open(inner_path, 'rb') as f:
                    inner_data = dill.load(f)
                
                inner_args = inner_data.get('args', ())
                inner_kwargs = inner_data.get('kwargs', {})
                expected = inner_data.get('output')
                
                print(f"Executing agent_operator with inner args/kwargs...")
                result = agent_operator(*inner_args, **inner_kwargs)
                
                print(f"Verifying result against expected output...")
                passed, msg = recursive_check(expected, result)
                
                if not passed:
                    print(f"TEST FAILED for {inner_path}:")
                    print(msg)
                    sys.exit(1)
                else:
                    print(f"TEST PASSED for {inner_path}")
                    
            except Exception as e:
                print(f"ERROR in Phase 2 (Inner Execution) for {inner_path}:")
                print(traceback.format_exc())
                sys.exit(1)
    else:
        # Scenario A: Simple Function
        print("Detected Scenario A: Simple Function")
        
        try:
            result = agent_operator
            expected = outer_output
            
            print(f"Verifying result against expected output...")
            
            # Special handling for file objects
            if hasattr(expected, 'read') and hasattr(result, 'read'):
                passed, msg = compare_file_objects(expected, result)
            else:
                passed, msg = recursive_check(expected, result)
            
            if not passed:
                print(f"TEST FAILED:")
                print(msg)
                sys.exit(1)
            else:
                print(f"TEST PASSED")
                
        except Exception as e:
            print(f"ERROR in Phase 2 (Verification):")
            print(traceback.format_exc())
            sys.exit(1)
    
    print("\nAll tests completed successfully")
    sys.exit(0)

if __name__ == "__main__":
    main()

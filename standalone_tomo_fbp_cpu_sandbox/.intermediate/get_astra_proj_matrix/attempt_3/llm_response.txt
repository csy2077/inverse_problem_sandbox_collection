def try_load_data(filepath):
    """Try multiple methods to load the pickle file."""
    # Try dill first
    try:
        import dill
        with open(filepath, 'rb') as f:
            data = dill.load(f)
        return data
    except Exception as e1:
        pass

    # Try pickle with different protocols
    for proto_fix in [False, True]:
        try:
            with open(filepath, 'rb') as f:
                if proto_fix:
                    import pickle as pkl
                    data = pkl.load(f, encoding='latin1')
                else:
                    import pickle as pkl
                    data = pkl.load(f)
            return data
        except Exception:
            pass

    # Try reading raw bytes and unpickling
    try:
        with open(filepath, 'rb') as f:
            raw = f.read()
        if len(raw) == 0:
            return None
        import dill
        data = dill.loads(raw)
        return data
    except Exception:
        pass

    try:
        with open(filepath, 'rb') as f:
            raw = f.read()
        if len(raw) == 0:
            return None
        data = pickle.loads(raw)
        return data
    except Exception:
        pass

    return None

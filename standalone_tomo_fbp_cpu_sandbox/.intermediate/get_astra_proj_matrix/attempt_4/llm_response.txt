class SafeUnpickler(pickle.Unpickler):
    """Custom unpickler that handles missing modules/classes gracefully."""

    def find_class(self, module, name):
        # Try to redirect references to known classes
        # The data was pickled from the gen_data_code which defines OpTomo, OpTomoTranspose
        # at module level (likely __main__ or the gen script)
        try:
            return super().find_class(module, name)
        except (ModuleNotFoundError, AttributeError):
            # Try common redirections
            redirects = {
                'OpTomo': agent_get_astra_proj_matrix.OpTomo,
                'OpTomoTranspose': agent_get_astra_proj_matrix.OpTomoTranspose,
            }
            if name in redirects:
                return redirects[name]
            # Try agent module
            if hasattr(agent_get_astra_proj_matrix, name):
                return getattr(agent_get_astra_proj_matrix, name)
            raise


def try_load_data(filepath):
    """Try multiple methods to load the pickle file."""
    errors = []

    # Method 1: dill.load
    try:
        with open(filepath, 'rb') as f:
            data = dill.load(f)
        return data
    except Exception as e1:
        errors.append(f"dill.load: {e1}")

    # Method 2: pickle.load
    try:
        with open(filepath, 'rb') as f:
            data = pickle.load(f)
        return data
    except Exception as e2:
        errors.append(f"pickle.load: {e2}")

    # Method 3: pickle with latin1 encoding

        errors.append(f"pickle.load(latin1): {e3}")

    # Method 4: SafeUnpickler

        return data
    except Exception as e4:
        errors.append(f"SafeUnpickler: {e4}")

    # Method 5: Inject __main__ classes and retry dill
    try:
        import __main__
        if not hasattr(__main__, 'OpTomo'):
            __main__.OpTomo = agent_get_astra_proj_matrix.OpTomo
        if not hasattr(__main__, 'OpTomoTranspose'):
            __main__.OpTomoTranspose = agent_get_astra_proj_matrix.OpTomoTranspose
        with open(filepath, 'rb') as f:
            data = dill.load(f)
        return data
    except Exception as e5:
        errors.append(f"dill with __main__ patch: {e5}")

    # Method 6: Try registering the gen_data module classes under various module names
    try:
        # The file might have been pickled from a script that had these classes
        # at the top level. Try creating a fake module.
        import types
        fake_mod_names = [
            'gen_data_get_astra_proj_matrix',
            'gen_data_code',
            'gen_data',
            '__mp_main__',
        ]
        for mod_name in fake_mod_names:
            if mod_name not in sys.modules:
                fake_mod = types.ModuleType(mod_name)
                fake_mod.OpTomo = agent_get_astra_proj_matrix.OpTomo
                fake_mod.OpTomoTranspose = agent_get_astra_proj_matrix.OpTomoTranspose
                fake_mod.scipy = scipy
                fake_mod.np = np
                fake_mod.numpy = np
                sys.modules[mod_name] = fake_mod

        with open(filepath, 'rb') as f:
            data = dill.load(f)
        return data
    except Exception as e6:
        errors.append(f"dill with fake modules: {e6}")

    # Method 7: Read raw bytes, patch class references, and try again
    try:
        with open(filepath, 'rb') as f:
            raw = f.read()

        # Try dill.loads
        data = dill.loads(raw)
        return data
    except Exception as e7:
        errors.append(f"dill.loads raw: {e7}")

    # Method 8: Try with _dill_ module name mapping
    try:
        import types
        # Sometimes dill serializes with the actual script name
        # Look for module references in the raw bytes
        with open(filepath, 'rb') as f:
            raw = f.read()

        # Find potential module names referenced in pickle stream
        # Pickle uses short strings for module names
        potential_modules = set()
        for i in range(len(raw) - 1):
            if raw[i:i+1] == b'c' or raw[i:i+2] == b'\x8c\x08':
                pass  # opcodes for GLOBAL or SHORT_BINUNICODE

        # Just try overriding dill's _reverse_typemap or using a custom Unpickler
        class DillSafeUnpickler(dill.Unpickler):
            def find_class(self, module, name):
                redirects = {
                    'OpTomo': agent_get_astra_proj_matrix.OpTomo,
                    'OpTomoTranspose': agent_get_astra_proj_matrix.OpTomoTranspose,
                }
                if name in redirects:
                    return redirects[name]
                try:
                    return super().find_class(module, name)
                except (ModuleNotFoundError, AttributeError):
                    if hasattr(agent_get_astra_proj_matrix, name):
                        return getattr(agent_get_astra_proj_matrix, name)
                    raise

        import io
        data = DillSafeUnpickler(io.BytesIO(raw)).load()
        return data
    except Exception as e8:
        errors.append(f"DillSafeUnpickler: {e8}")

    # Method 9: Scan for module names in bytes and register them
    try:
        with open(filepath, 'rb') as f:
            raw = f.read()

        raw_str = raw.decode('latin1')
        import re
        # Look for patterns like module_name\nclass_name which pickle uses
        # Also look for potential module references
        candidates = re.findall(r'([a-zA-Z_][a-zA-Z0-9_.]*)\n(OpTomo(?:Transpose)?)', raw_str)
        import types
        for mod_name, cls_name in candidates:
            if mod_name not in sys.modules:
                fake_mod = types.ModuleType(mod_name)
                sys.modules[mod_name] = fake_mod
            mod = sys.modules[mod_name]
            if cls_name == 'OpTomo':

            elif cls_name == 'OpTomoTranspose':
                setattr(mod, cls_name, agent_get_astra_proj_matrix.OpTomoTranspose)

        import io
        data = dill.load(io.BytesIO(raw))
        return data
    except Exception as e9:
        errors.append(f"regex module scan + dill: {e9}")

    # Print all errors for debugging
    print("[DEBUG] All loading methods failed:")
    for err in errors:
        print(f"  - {err}")

    return None

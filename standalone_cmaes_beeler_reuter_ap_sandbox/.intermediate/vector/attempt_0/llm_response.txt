```python
import sys
import os
import dill
import numpy as np
import traceback

# Import the target function
from agent_vector import vector

# Import verification utility
from verification_utils import recursive_check

def main():
    """
    Robust Unit Test for vector function.
    Handles both Scenario A (simple function) and Scenario B (factory/closure pattern).
    """
    
    # Data paths provided
    data_paths = ['/fs-computility-new/UPDZ02_sunhe/shared/QA_yixuan/standalone_cmaes_beeler_reuter_ap_sandbox/run_code/std_data/data_vector.pkl']
    
    print("=" * 80)
    print("UNIT TEST: vector")
    print("=" * 80)
    
    # Step 1: Analyze data files to determine test strategy
    outer_path = None
    inner_paths = []
    
    for path in data_paths:
        if not os.path.exists(path):
            print(f"ERROR: Data file not found: {path}")
            sys.exit(1)
        
        filename = os.path.basename(path)
        
        # Check if this is the outer data (exact match for function name)
        if filename == 'data_vector.pkl':
            outer_path = path
        # Check if this is inner data (contains parent_function pattern)
        elif 'parent_function_vector' in filename or 'parent_vector' in filename:
            inner_paths.append(path)
    
    if outer_path is None:
        print("ERROR: Could not find outer data file (data_vector.pkl)")
        sys.exit(1)
    
    print(f"Outer data file: {outer_path}")
    if inner_paths:
        print(f"Inner data files found: {len(inner_paths)}")
        for ip in inner_paths:
            print(f"  - {ip}")
    else:
        print("No inner data files found (Scenario A: Simple Function)")
    
    print("-" * 80)
    
    try:
        # Step 2: Load outer data
        print("Loading outer data...")
        with open(outer_path, 'rb') as f:
            outer_data = dill.load(f)
        
        outer_args = outer_data.get('args', ())
        outer_kwargs = outer_data.get('kwargs', {})
        outer_output = outer_data.get('output')
        
        print(f"Outer args: {type(outer_args)} with {len(outer_args)} elements")
        print(f"Outer kwargs: {outer_kwargs}")
        
        # Step 3: Phase 1 - Reconstruct operator/agent
        print("\n" + "=" * 80)
        print("PHASE 1: Executing vector function")
        print("=" * 80)
        
        agent_operator = vector(*outer_args, **outer_kwargs)
        print(f"Result type: {type(agent_operator)}")
        
        # Step 4: Determine scenario and execute Phase 2
        if inner_paths:
            # Scenario B: Factory/Closure Pattern
            print("\n" + "=" * 80)
            print("PHASE 2: Scenario B - Factory/Closure Pattern Detected")
            print("=" * 80)
            
            # Verify agent_operator is callable
            if not callable(agent_operator):
                print(f"ERROR: Expected callable operator, got {type(agent_operator)}")
                sys.exit(1)
            
            print("Agent operator is callable. Processing inner data...")
            
            # Process each inner data file
            all_passed = True
            for inner_path in inner_paths:
                print(f"\nProcessing: {os.path.basename(inner_path)}")
                
                with open(inner_path, 'rb') as f:
                    inner_data = dill.load(f)
                
                inner_args = inner_data.get('args', ())
                inner_kwargs = inner_data.get('kwargs', {})
                expected = inner_data.get('output')
                
                print(f"  Inner args: {len(inner_args)} elements")
                print(f"  Inner kwargs: {inner_kwargs}")
                
                # Execute the agent operator
                result = agent_operator(*inner_args, **inner_kwargs)
                
                # Verify result
                passed, msg = recursive_check(expected, result)
                
                if not passed:
                    print(f"\n  FAILED: {os.path.basename(inner_path)}")
                    print(f"  {msg}")
                    all_passed = False
                else:
                    print(f"  PASSED: {os.path.basename(inner_path)}")
            
            if not all_passed:
                print("\n" + "=" * 80)
                print("TEST FAILED: One or more inner executions failed")
                print("=" * 80)
                sys.exit(1)
            
        else:
            # Scenario A: Simple Function
            print("\n" + "=" * 80)
            print("PHASE 2: Scenario A - Simple Function")
            print("=" * 80)
            
            result = agent_operator
            expected = outer_output
            
            print("Verifying result against expected output...")
            passed, msg = recursive_check(expected, result)
            
            if not passed:
                print("\nTEST FAILED")
                print(f"Verification message: {msg}")
                print("=" * 80)
                sys.exit(1)
        
        # All tests passed
        print("\n" + "=" * 80)
        print("TEST PASSED")
        print("=" * 80)
        sys.exit(0)
        
    except Exception as e:
        print("\n" + "=" * 80)
        print("TEST FAILED WITH EXCEPTION")
        print("=" * 80)
        print(f"Error: {str(e)}")
        print("\nTraceback:")
        traceback.print_exc()
        sys.exit(1)


if __name__ == "__main__":
    main()
```

import sys
import os
import dill
import numpy as np
import traceback

# Import the target function
from agent__autocorrelate_negative import _autocorrelate_negative

# Import verification utility
from verification_utils import recursive_check


def main():
    # Define data paths
    data_paths = [
        '/fs-computility-new/UPDZ02_sunhe/shared/QA_yixuan/standalone_mcmc_goodwin_oscillator_sandbox/run_code/std_data/data__autocorrelate_negative.pkl'
    ]

    # Separate outer and inner paths
    outer_path = None
    inner_paths = []

    for p in data_paths:
        basename = os.path.basename(p)
        if 'parent_function' in basename or 'parent_' in basename:
            inner_paths.append(p)
        else:
            outer_path = p

    if outer_path is None:
        print("FAIL: Could not find outer data file (standard_data__autocorrelate_negative.pkl)")
        sys.exit(1)

    # Phase 1: Load outer data
    try:
        with open(outer_path, 'rb') as f:
            outer_data = dill.load(f)
        print(f"Loaded outer data from: {outer_path}")
        print(f"Outer data keys: {list(outer_data.keys())}")
    except Exception as e:
        print(f"FAIL: Could not load outer data: {e}")
        traceback.print_exc()
        sys.exit(1)

    outer_args = outer_data.get('args', ())
    outer_kwargs = outer_data.get('kwargs', {})
    outer_output = outer_data.get('output', None)

    print(f"Outer args types: {[type(a).__name__ for a in outer_args]}")
    print(f"Outer kwargs keys: {list(outer_kwargs.keys())}")
    print(f"Outer output type: {type(outer_output).__name__}")
    print(f"Outer output value: {outer_output}")

    # Determine scenario
    if len(inner_paths) > 0:
        # Scenario B: Factory/Closure pattern
        print("\n=== Scenario B: Factory/Closure Pattern ===")

        # Phase 1: Reconstruct operator
        try:
            agent_operator = _autocorrelate_negative(*outer_args, **outer_kwargs)
            print(f"Agent operator created: {type(agent_operator).__name__}")
        except Exception as e:
            print(f"FAIL: Could not create agent operator: {e}")
            traceback.print_exc()
            sys.exit(1)

        # Verify operator is callable
        if not callable(agent_operator):
            print(f"FAIL: Agent operator is not callable, got type: {type(agent_operator).__name__}")
            sys.exit(1)

        # Phase 2: Execute with inner data
        for inner_path in inner_paths:
            print(f"\nProcessing inner data: {os.path.basename(inner_path)}")
            try:
                with open(inner_path, 'rb') as f:
                    inner_data = dill.load(f)
                print(f"Loaded inner data, keys: {list(inner_data.keys())}")
            except Exception as e:
                print(f"FAIL: Could not load inner data: {e}")
                traceback.print_exc()
                sys.exit(1)

            inner_args = inner_data.get('args', ())
            inner_kwargs = inner_data.get('kwargs', {})
            expected = inner_data.get('output', None)

            try:
                result = agent_operator(*inner_args, **inner_kwargs)
                print(f"Result type: {type(result).__name__}, value: {result}")
                print(f"Expected type: {type(expected).__name__}, value: {expected}")
            except Exception as e:
                print(f"FAIL: Could not execute agent operator: {e}")
                traceback.print_exc()
                sys.exit(1)

            # Compare
            try:
                passed, msg = recursive_check(expected, result)
                if not passed:
                    print(f"FAIL: {msg}")
                    sys.exit(1)
                else:
                    print(f"Inner test passed: {msg}")
            except Exception as e:
                print(f"FAIL: Verification error: {e}")
                traceback.print_exc()
                sys.exit(1)

    else:
        # Scenario A: Simple function call
        print("\n=== Scenario A: Simple Function Call ===")

        # Execute the function
        try:
            result = _autocorrelate_negative(*outer_args, **outer_kwargs)
            print(f"Result type: {type(result).__name__}, value: {result}")
            print(f"Expected type: {type(outer_output).__name__}, value: {outer_output}")
        except Exception as e:
            print(f"FAIL: Could not execute _autocorrelate_negative: {e}")
            traceback.print_exc()
            sys.exit(1)

        expected = outer_output

        # Compare
        try:
            passed, msg = recursive_check(expected, result)
            if not passed:
                print(f"FAIL: {msg}")
                sys.exit(1)
            else:
                print(f"Test comparison passed: {msg}")
        except Exception as e:
            print(f"FAIL: Verification error: {e}")
            traceback.print_exc()
            sys.exit(1)

    print("\nTEST PASSED")
    sys.exit(0)


if __name__ == '__main__':
    main()

```python
import sys
import os
import dill
import traceback
import logging

# Ensure the current directory is in sys.path for imports
sys.path.insert(0, os.path.dirname(os.path.abspath(__file__)))

from agent_create_logger import create_logger
from verification_utils import recursive_check


def main():
    data_paths = [
        '/fs-computility-new/UPDZ02_sunhe/shared/QA_yixuan/standalone_mri_csgm_mri_sandbox/run_code/std_data/data_create_logger.pkl'
    ]

    # Separate outer and inner paths
    outer_path = None
    inner_paths = []

    for p in data_paths:
        basename = os.path.basename(p)
        if 'parent_function' in basename or 'parent_' in basename:
            inner_paths.append(p)
        else:
            outer_path = p

    if outer_path is None:
        print("FAIL: No outer data file found for create_logger.")
        sys.exit(1)

    # Phase 1: Load outer data and reconstruct operator
    try:
        with open(outer_path, 'rb') as f:
            outer_data = dill.load(f)
    except Exception as e:
        print(f"FAIL: Could not load outer data from {outer_path}")
        traceback.print_exc()
        sys.exit(1)

    outer_args = outer_data.get('args', ())
    outer_kwargs = outer_data.get('kwargs', {})
    outer_output = outer_data.get('output', None)

    print(f"Loaded outer data: func_name={outer_data.get('func_name')}")
    print(f"  args types: {[type(a).__name__ for a in outer_args]}")
    print(f"  kwargs keys: {list(outer_kwargs.keys())}")

    # Inspect the expected output to understand what the data capture recorded
    if isinstance(outer_output, logging.Logger):
        print(f"  expected logger name: {outer_output.name}")
        print(f"  expected logger level: {outer_output.level}")
        print(f"  expected logger handlers: {[type(h).__name__ for h in outer_output.handlers]}")

    # Before calling create_logger, ensure the logging_dir exists so FileHandler works
    logging_dir = None
    if len(outer_args) > 0:
        logging_dir = outer_args[0]
        if isinstance(logging_dir, str):
            os.makedirs(logging_dir, exist_ok=True)
    if 'logging_dir' in outer_kwargs:
        logging_dir = outer_kwargs['logging_dir']
        if isinstance(logging_dir, str):
            os.makedirs(logging_dir, exist_ok=True)

    # The gen_data_code uses the same module-level logger (__name__ in gen context).
    # The data was captured via the decorator which calls func() then saves the result.
    # The captured logger's state reflects what it looked like RIGHT AFTER creation,
    # but the logger object is a singleton - by the time it was pickled, handlers may
    # have been cleared or the logger state changed.
    #
    # The key insight from the error:
    #   expected level=0 (NOTSET), expected handlers=[]
    # This means the pickled logger lost its state (loggers don't pickle well).
    # 
    # So we need to verify the logger was created correctly by checking its properties
    # directly rather than comparing against the (broken) pickled version.

    # Clear existing handlers on the module logger to ensure a clean state
    # The logger name in agent_create_logger.py uses __name__ which is 'agent_create_logger'
    module_logger_name = 'agent_create_logger'
    existing_logger = logging.getLogger(module_logger_name)
    existing_logger.handlers.clear()
    existing_logger.setLevel(logging.WARNING)  # Reset to default

    try:
        agent_result = create_logger(*outer_args, **outer_kwargs)
    except Exception as e:
        print(f"FAIL: create_logger raised an exception: {e}")
        traceback.print_exc()
        sys.exit(1)

    print(f"  agent_result type: {type(agent_result).__name__}")

    # Phase 2: Check if there are inner paths (Scenario B) or not (Scenario A)
    if inner_paths:
        # Scenario B: Factory/Closure pattern
        if not callable(agent_result):
            print(f"FAIL: Expected callable from create_logger, got {type(agent_result).__name__}")
            sys.exit(1)

        for inner_path in inner_paths:
            try:
                with open(inner_path, 'rb') as f:
                    inner_data = dill.load(f)
            except Exception as e:
                print(f"FAIL: Could not load inner data from {inner_path}")
                traceback.print_exc()
                sys.exit(1)

            inner_args = inner_data.get('args', ())
            inner_kwargs = inner_data.get('kwargs', {})
            expected = inner_data.get('output', None)

            print(f"Loaded inner data: func_name={inner_data.get('func_name')}")
            print(f"  inner_args types: {[type(a).__name__ for a in inner_args]}")
            print(f"  inner_kwargs keys: {list(inner_kwargs.keys())}")

            try:
                actual_result = agent_result(*inner_args, **inner_kwargs)
            except Exception as e:
                print(f"FAIL: Executing agent_operator raised an exception: {e}")
                traceback.print_exc()
                sys.exit(1)

            try:
                passed, msg = recursive_check(expected, actual_result)
            except Exception as e:
                print(f"FAIL: recursive_check raised an exception: {e}")
                traceback.print_exc()
                sys.exit(1)

            if not passed:
                print(f"FAIL: Verification failed for inner data {os.path.basename(inner_path)}")
                print(f"  Message: {msg}")
                sys.exit(1)

            print(f"  Inner test passed for {os.path.basename(inner_path)}")
    else:
        # Scenario A: Simple function - result is a Logger
        # Logger objects don't pickle/unpickle well - the deserialized expected logger
        # has level=0 and no handlers. Instead of comparing against the broken pickled
        # version, we verify the logger's properties based on the function's contract.

        if not isinstance(agent_result, logging.Logger):
            print(f"FAIL: Expected logging.Logger instance, got {type(agent_result).__name__}")
            sys.exit(1)

        checks_passed = True
        fail_messages = []

        # Determine which branch was taken based on the kwargs
        main_process = outer_kwargs.get('main_process', True)
        if len(outer_args) > 1:
            main_process = outer_args[1]

        if main_process:
            # main_process=True branch: level should be INFO, should have StreamHandler and FileHandler
            if agent_result.level != logging.INFO:
                checks_passed = False
                fail_messages.append(
                    f"Logger level mismatch: expected {logging.INFO} (INFO), got {agent_result.level}"
                )

            actual_handler_types = sorted([type(h).__name__ for h in agent_result.handlers])
            expected_handler_types = sorted(['FileHandler', 'StreamHandler'])
            if actual_handler_types != expected_handler_types:
                checks_passed = False
                fail_messages.append(
                    f"Handler types mismatch: expected {expected_handler_types}, got {actual_handler_types}"
                )

            # Verify FileHandler points to the right file
            file_handlers = [h for h in agent_result.handlers if isinstance(h, logging.FileHandler)]
            if file_handlers and logging_dir:
                expected_log_path = os.path.join(logging_dir, 'log.txt')
                actual_log_path = file_handlers[0].baseFilename
                if os.path.abspath(expected_log_path) != os.path.abspath(actual_log_path):
                    checks_passed = False
                    fail_messages.append(
                        f"FileHandler path mismatch: expected {os.path.abspath(expected_log_path)}, got {actual_log_path}"
                    )
        else:
            # main_process=False branch: should have NullHandler
            has_null = any(isinstance(h, logging.NullHandler) for h in agent_result.handlers)
            if not has_null:
                checks_passed = False
                fail_messages.append("Expected NullHandler for non-main process, not found")

        if not checks_passed:
            for msg in fail_messages:
                print(f"FAIL: {msg}")
            sys.exit(1)

    print("TEST PASSED")
    sys.exit(0)


if __name__ == '__main__':
    main()
```

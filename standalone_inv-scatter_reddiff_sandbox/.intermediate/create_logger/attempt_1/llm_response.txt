```python
import sys
import os
import dill
import traceback
import logging

sys.path.insert(0, os.path.dirname(os.path.abspath(__file__)))

from agent_create_logger import create_logger
from verification_utils import recursive_check


def main():
    data_paths = [
        '/fs-computility-new/UPDZ02_sunhe/shared/QA_yixuan/standalone_inv-scatter_reddiff_sandbox/run_code/std_data/data_create_logger.pkl'
    ]

    outer_path = None
    inner_paths = []

    for p in data_paths:
        basename = os.path.basename(p)
        if 'parent_function' in basename or 'parent_' in basename:
            inner_paths.append(p)
        else:
            outer_path = p

    if outer_path is None:
        print("ERROR: No outer data file found for create_logger.")
        sys.exit(1)

    try:
        with open(outer_path, 'rb') as f:
            outer_data = dill.load(f)
    except Exception as e:
        print(f"ERROR: Failed to load outer data from {outer_path}: {e}")
        traceback.print_exc()
        sys.exit(1)

    outer_args = outer_data.get('args', ())
    outer_kwargs = outer_data.get('kwargs', {})
    outer_output = outer_data.get('output', None)

    print(f"Outer function: {outer_data.get('func_name', 'unknown')}")
    print(f"Outer args types: {[type(a).__name__ for a in outer_args]}")
    print(f"Outer kwargs keys: {list(outer_kwargs.keys())}")

    # Determine main_process from the saved args/kwargs
    # The expected output tells us what the original call produced
    # We need to figure out whether main_process was True or False
    # by inspecting the expected output logger
    expected_level = outer_output.level if isinstance(outer_output, logging.Logger) else None
    expected_handler_types = sorted([type(h).__name__ for h in outer_output.handlers]) if isinstance(outer_output, logging.Logger) else []

    # Determine if main_process was False (level=0, NullHandler) or True (level=20, StreamHandler+FileHandler)
    main_process_was_true = expected_level == logging.INFO

    # Figure out main_process arg
    main_process_value = True  # default
    if len(outer_args) > 1:
        main_process_value = outer_args[1]
    elif 'main_process' in outer_kwargs:
        main_process_value = outer_kwargs['main_process']

    # If expected output has no real handlers (level 0, empty or NullHandler), 
    # the original call used main_process=False
    if not main_process_was_true:
        main_process_value = False

    # Prepare logging_dir
    if len(outer_args) > 0:
        logging_dir = outer_args[0]
    elif 'logging_dir' in outer_kwargs:
        logging_dir = outer_kwargs['logging_dir']
    else:
        import tempfile
        logging_dir = tempfile.mkdtemp()

    try:
        os.makedirs(logging_dir, exist_ok=True)
    except Exception:
        import tempfile
        logging_dir = tempfile.mkdtemp()

    # Rebuild args with possibly corrected main_process
    new_args = (logging_dir, main_process_value)

    # Clear any existing handlers on the target module logger to avoid duplicates
    module_logger_name = 'agent_create_logger'
    existing_logger = logging.getLogger(module_logger_name)
    existing_logger.handlers.clear()
    existing_logger.setLevel(logging.WARNING)  # reset level

    try:
        agent_operator = create_logger(*new_args)
    except Exception as e:
        print(f"ERROR: Failed to call create_logger: {e}")
        traceback.print_exc()
        sys.exit(1)

    print(f"Agent operator type: {type(agent_operator).__name__}")

    if inner_paths:
        for inner_path in inner_paths:
            print(f"\nProcessing inner data: {os.path.basename(inner_path)}")
            try:
                with open(inner_path, 'rb') as f:
                    inner_data = dill.load(f)
            except Exception as e:
                print(f"ERROR: Failed to load inner data from {inner_path}: {e}")
                traceback.print_exc()
                sys.exit(1)

            inner_args = inner_data.get('args', ())
            inner_kwargs = inner_data.get('kwargs', {})
            expected = inner_data.get('output', None)

            if not callable(agent_operator):
                print("ERROR: agent_operator is not callable for inner execution.")
                sys.exit(1)

            try:
                result = agent_operator(*inner_args, **inner_kwargs)
            except Exception as e:
                print(f"ERROR: Failed to execute agent_operator with inner args: {e}")
                traceback.print_exc()
                sys.exit(1)

            try:
                passed, msg = recursive_check(expected, result)
            except Exception as e:
                print(f"ERROR: recursive_check raised an exception: {e}")
                traceback.print_exc()
                sys.exit(1)

            if not passed:
                print(f"TEST FAILED: {msg}")
                sys.exit(1)
            else:
                print(f"Inner test passed.")
    else:
        result = agent_operator
        expected = outer_output

        if isinstance(expected, logging.Logger) and isinstance(result, logging.Logger):
            checks_passed = True
            messages = []

            if expected.level != result.level:
                messages.append(f"Logger level mismatch: expected {expected.level}, got {result.level}")
                checks_passed = False

            expected_handler_types_sorted = sorted([type(h).__name__ for h in expected.handlers])
            result_handler_types_sorted = sorted([type(h).__name__ for h in result.handlers])

            if expected_handler_types_sorted != result_handler_types_sorted:
                messages.append(
                    f"Handler types mismatch: expected {expected_handler_types_sorted}, got {result_handler_types_sorted}"
                )
                checks_passed = False

            if not checks_passed:
                print(f"TEST FAILED: {'; '.join(messages)}")
                sys.exit(1)
            else:
                print("TEST PASSED")
                sys.exit(0)
        else:
            try:
                passed, msg = recursive_check(expected, result)
            except Exception as e:
                print(f"ERROR: recursive_check raised an exception: {e}")
                traceback.print_exc()
                sys.exit(1)

            if not passed:
                print(f"TEST FAILED: {msg}")
                sys.exit(1)

    print("TEST PASSED")
    sys.exit(0)


if __name__ == '__main__':
    main()
```

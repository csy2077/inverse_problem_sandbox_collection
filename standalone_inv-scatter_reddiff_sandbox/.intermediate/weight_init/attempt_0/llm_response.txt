import sys
import os
import dill
import torch
import numpy as np
import traceback

# Import the target function
from agent_weight_init import weight_init

# Import verification utility
from verification_utils import recursive_check


def main():
    # All data paths provided
    data_paths = [
        '/fs-computility-new/UPDZ02_sunhe/shared/QA_yixuan/standalone_inv-scatter_reddiff_sandbox/run_code/std_data/data_weight_init.pkl'
    ]

    # Separate outer (direct function data) and inner (parent_function / closure) paths
    outer_path = None
    inner_paths = []

    for p in data_paths:
        basename = os.path.basename(p)
        if 'parent_function' in basename or 'parent_' in basename:
            inner_paths.append(p)
        else:
            outer_path = p

    if outer_path is None:
        print("FAIL: Could not find outer data file (data_weight_init.pkl).")
        sys.exit(1)

    # ---------------------------------------------------------------
    # Phase 1: Load outer data and reconstruct / execute the function
    # ---------------------------------------------------------------
    try:
        with open(outer_path, 'rb') as f:
            outer_data = dill.load(f)
    except Exception as e:
        print(f"FAIL: Could not load outer data from {outer_path}: {e}")
        traceback.print_exc()
        sys.exit(1)

    outer_args = outer_data.get('args', ())
    outer_kwargs = outer_data.get('kwargs', {})
    outer_output = outer_data.get('output', None)

    print(f"[INFO] Loaded outer data: func_name={outer_data.get('func_name')}")
    print(f"[INFO] outer_args types: {[type(a).__name__ for a in outer_args]}")
    print(f"[INFO] outer_kwargs keys: {list(outer_kwargs.keys())}")

    # NOTE: weight_init uses torch.rand / torch.randn internally, which are stochastic.
    # The function's output depends on random state, so we cannot reproduce the exact
    # same tensor values unless we know the random seed.
    # However, we CAN verify:
    #   1. The output shape matches.
    #   2. The output dtype matches.
    #   3. The output is a tensor.
    #   4. The scale (std / range) is approximately correct for the given mode.
    # But first, let's try recursive_check — if the test data was captured with a
    # specific seed and we're meant to do statistical/structural checks, recursive_check
    # with tolerance should handle it.

    # Since weight_init is stochastic (uses torch.rand/torch.randn), we verify
    # structural properties rather than exact values.

    try:
        # Execute the function
        actual_result = weight_init(*outer_args, **outer_kwargs)
    except Exception as e:
        print(f"FAIL: weight_init raised an exception: {e}")
        traceback.print_exc()
        sys.exit(1)

    # ---------------------------------------------------------------
    # Phase 2: Check if there are inner paths (closure/factory pattern)
    # ---------------------------------------------------------------
    if inner_paths:
        # Scenario B: The result should be callable
        if not callable(actual_result):
            print(f"FAIL: Expected weight_init to return a callable (closure), got {type(actual_result)}")
            sys.exit(1)

        for inner_path in inner_paths:
            try:
                with open(inner_path, 'rb') as f:
                    inner_data = dill.load(f)
            except Exception as e:
                print(f"FAIL: Could not load inner data from {inner_path}: {e}")
                traceback.print_exc()
                sys.exit(1)

            inner_args = inner_data.get('args', ())
            inner_kwargs = inner_data.get('kwargs', {})
            expected = inner_data.get('output', None)

            print(f"[INFO] Loaded inner data: func_name={inner_data.get('func_name')}")

            try:
                result = actual_result(*inner_args, **inner_kwargs)
            except Exception as e:
                print(f"FAIL: Calling the operator raised an exception: {e}")
                traceback.print_exc()
                sys.exit(1)

            try:
                passed, msg = recursive_check(expected, result)
            except Exception as e:
                print(f"FAIL: recursive_check raised an exception: {e}")
                traceback.print_exc()
                sys.exit(1)

            if not passed:
                print(f"FAIL: Verification failed for inner data {inner_path}")
                print(f"Message: {msg}")
                sys.exit(1)
            else:
                print(f"[PASS] Inner test passed for {os.path.basename(inner_path)}")

    else:
        # Scenario A: Simple function — the result IS the output.
        # weight_init is stochastic, so exact match will fail.
        # We do structural checks instead.
        expected = outer_output

        # First, try recursive_check in case it handles tolerance / structural matching
        try:
            passed, msg = recursive_check(expected, actual_result)
        except Exception as e:
            print(f"FAIL: recursive_check raised an exception: {e}")
            traceback.print_exc()
            sys.exit(1)

        if passed:
            print("[PASS] recursive_check passed (exact or tolerant match).")
        else:
            # Since weight_init is stochastic, fall back to structural verification
            print(f"[INFO] recursive_check did not pass (expected for stochastic function): {msg}")
            print("[INFO] Falling back to structural verification...")

            structural_pass = True
            fail_reasons = []

            # Check type
            if type(expected) != type(actual_result):
                structural_pass = False
                fail_reasons.append(
                    f"Type mismatch: expected {type(expected)}, got {type(actual_result)}"
                )

            # Check shape for tensors
            if isinstance(expected, torch.Tensor) and isinstance(actual_result, torch.Tensor):
                if expected.shape != actual_result.shape:
                    structural_pass = False
                    fail_reasons.append(
                        f"Shape mismatch: expected {expected.shape}, got {actual_result.shape}"
                    )
                if expected.dtype != actual_result.dtype:
                    structural_pass = False
                    fail_reasons.append(
                        f"Dtype mismatch: expected {expected.dtype}, got {actual_result.dtype}"
                    )

                # Verify the scale is reasonable based on the mode
                mode = None
                fan_in = None
                fan_out = None

                # Extract mode, fan_in, fan_out from outer_args/kwargs
                try:
                    import inspect
                    sig = inspect.signature(weight_init)
                    param_names = list(sig.parameters.keys())
                    bound = sig.bind(*outer_args, **outer_kwargs)
                    bound.apply_defaults()
                    mode = bound.arguments.get('mode')
                    fan_in = bound.arguments.get('fan_in')
                    fan_out = bound.arguments.get('fan_out')
                except Exception:
                    pass

                if mode and fan_in is not None:
                    actual_std = actual_result.float().std().item()
                    expected_std = expected.float().std().item()

                    # For stochastic outputs, we check that the standard deviation
                    # is in a reasonable ballpark (within 5x of expected).
                    # This is a very loose check just to ensure the right scaling.
                    if expected_std > 0:
                        ratio = actual_std / expected_std
                        if ratio < 0.1 or ratio > 10.0:
                            structural_pass = False
                            fail_reasons.append(
                                f"Scale mismatch: expected std ~{expected_std:.4f}, "
                                f"got {actual_std:.4f} (ratio={ratio:.2f})"
                            )
                    print(f"[INFO] Mode: {mode}, expected_std: {expected_std:.4f}, actual_std: {actual_std:.4f}")

            elif isinstance(expected, np.ndarray) and isinstance(actual_result, np.ndarray):
                if expected.shape != actual_result.shape:
                    structural_pass = False
                    fail_reasons.append(
                        f"Shape mismatch: expected {expected.shape}, got {actual_result.shape}"
                    )

            if structural_pass:
                print("[PASS] Structural verification passed (shape, dtype, scale).")
            else:
                print("FAIL: Structural verification failed.")
                for reason in fail_reasons:
                    print(f"  - {reason}")
                sys.exit(1)

    print("TEST PASSED")
    sys.exit(0)


if __name__ == '__main__':
    main()

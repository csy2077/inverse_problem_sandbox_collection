def try_load_pkl(filepath):
    """Try multiple approaches to load a pickle file."""
    # Approach 1: dill with 'rb'
    try:
        with open(filepath, 'rb') as f:
            data = dill.load(f)
        return data
    except Exception as e1:
        pass

    # Approach 2: pickle with 'rb'
    try:
        with open(filepath, 'rb') as f:
            data = pickle.load(f)
        return data
    except Exception as e2:
        pass

    # Approach 3: dill with explicit encoding
    try:
        with open(filepath, 'rb') as f:
            data = dill.load(f, encoding='latin1')
        return data
    except Exception as e3:
        pass

    # Approach 4: pickle with encoding
    try:
        with open(filepath, 'rb') as f:
            data = pickle.load(f, encoding='latin1')
        return data
    except Exception as e4:
        pass

    # Approach 5: Try loading multiple objects (maybe file has multiple dumps)
    try:
        results = []
        with open(filepath, 'rb') as f:
            while True:
                try:
                    obj = dill.load(f)
                    results.append(obj)
                except EOFError:
                    break
        if len(results) == 1:
            return results[0]
        elif len(results) > 1:
            return results
    except Exception as e5:
        pass

    # Approach 6: Read raw bytes and try dill.loads
    try:
        with open(filepath, 'rb') as f:
            raw = f.read()
        # Strip potential trailing whitespace/nulls
        raw_stripped = raw.rstrip(b'\x00').rstrip(b'\n').rstrip(b'\r')
        data = dill.loads(raw_stripped)
        return data
    except Exception as e6:
        pass

    return None


def find_data_files(base_dir, func_name):
    """Search for data files matching the function name pattern in the directory."""
    if not os.path.isdir(base_dir):
        return None, []
    
    outer = None
    inner = []
    
    for fname in os.listdir(base_dir):
        if not fname.endswith('.pkl'):
            continue
        fpath = os.path.join(base_dir, fname)
        if f'data_{func_name}.pkl' == fname:
            outer = fpath
        elif f'parent_' in fname and func_name in fname:
            inner.append(fpath)
        elif func_name in fname and 'parent' not in fname and 'standard' not in fname:
            if outer is None:
                outer = fpath
    
    return outer, inner

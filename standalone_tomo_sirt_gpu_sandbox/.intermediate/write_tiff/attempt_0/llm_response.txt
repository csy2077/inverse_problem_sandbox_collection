import sys
import os
import dill
import traceback
import numpy as np

# Ensure the working directory is on the path
sys.path.insert(0, os.path.dirname(os.path.abspath(__file__)))

from agent_write_tiff import write_tiff
from verification_utils import recursive_check

def main():
    data_paths = [
        '/fs-computility-new/UPDZ02_sunhe/shared/QA_yixuan/standalone_tomo_sirt_gpu_sandbox/run_code/std_data/data_write_tiff.pkl'
    ]

    # -------------------------------------------------------------------------
    # Step 1: Classify data files into outer (standard) and inner (parent) paths
    # -------------------------------------------------------------------------
    outer_path = None
    inner_paths = []

    for p in data_paths:
        basename = os.path.basename(p)
        if 'parent_function' in basename or 'parent_' in basename:
            inner_paths.append(p)
        else:
            outer_path = p

    if outer_path is None:
        print("FAIL: Could not find outer data file (standard_data / data_write_tiff.pkl).")
        sys.exit(1)

    # -------------------------------------------------------------------------
    # Step 2: Load outer data
    # -------------------------------------------------------------------------
    try:
        with open(outer_path, 'rb') as f:
            outer_data = dill.load(f)
        print(f"Loaded outer data from: {outer_path}")
        print(f"  Keys: {list(outer_data.keys()) if isinstance(outer_data, dict) else type(outer_data)}")
    except Exception as e:
        print(f"FAIL: Could not load outer data: {e}")
        traceback.print_exc()
        sys.exit(1)

    outer_args = outer_data.get('args', ())
    outer_kwargs = outer_data.get('kwargs', {})
    outer_output = outer_data.get('output', None)

    # -------------------------------------------------------------------------
    # Step 3: Determine scenario
    # -------------------------------------------------------------------------
    if len(inner_paths) > 0:
        # ----- Scenario B: Factory / Closure -----
        print("Detected Scenario B (Factory/Closure pattern).")

        # Phase 1: Reconstruct operator
        try:
            agent_operator = write_tiff(*outer_args, **outer_kwargs)
            print(f"  Phase 1 result type: {type(agent_operator)}")
        except Exception as e:
            print(f"FAIL: Phase 1 (operator creation) raised an exception: {e}")
            traceback.print_exc()
            sys.exit(1)

        if not callable(agent_operator):
            print(f"FAIL: Phase 1 result is not callable: {type(agent_operator)}")
            sys.exit(1)

        # Phase 2: Execute with inner data
        for inner_path in inner_paths:
            try:
                with open(inner_path, 'rb') as f:
                    inner_data = dill.load(f)
                print(f"Loaded inner data from: {inner_path}")
            except Exception as e:
                print(f"FAIL: Could not load inner data: {e}")
                traceback.print_exc()
                sys.exit(1)

            inner_args = inner_data.get('args', ())
            inner_kwargs = inner_data.get('kwargs', {})
            expected = inner_data.get('output', None)

            try:
                result = agent_operator(*inner_args, **inner_kwargs)
            except Exception as e:
                print(f"FAIL: Phase 2 (operator execution) raised an exception: {e}")
                traceback.print_exc()
                sys.exit(1)

            # Comparison
            try:
                passed, msg = recursive_check(expected, result)
            except Exception as e:
                print(f"FAIL: recursive_check raised an exception: {e}")
                traceback.print_exc()
                sys.exit(1)

            if not passed:
                print(f"FAIL: Verification failed for inner data {os.path.basename(inner_path)}.")
                print(f"  Message: {msg}")
                sys.exit(1)
            else:
                print(f"  Inner test passed for {os.path.basename(inner_path)}.")

    else:
        # ----- Scenario A: Simple function call -----
        print("Detected Scenario A (Simple function call).")

        # write_tiff returns None (it writes a file and prints).
        # We need to handle the side-effect nature of this function.
        # The function saves a TIFF file and returns None.

        try:
            result = write_tiff(*outer_args, **outer_kwargs)
            print(f"  Function returned: {result} (type: {type(result)})")
        except Exception as e:
            print(f"FAIL: Function execution raised an exception: {e}")
            traceback.print_exc()
            sys.exit(1)

        expected = outer_output

        # Comparison
        try:
            passed, msg = recursive_check(expected, result)
        except Exception as e:
            print(f"FAIL: recursive_check raised an exception: {e}")
            traceback.print_exc()
            sys.exit(1)

        if not passed:
            print(f"FAIL: Verification failed.")
            print(f"  Message: {msg}")

            # Additional verification: check that the file was actually created
            # Extract fname from args
            if len(outer_args) > 0:
                fname = outer_args[0]
                if not (fname.endswith('.tif') or fname.endswith('.tiff')):
                    fname = fname + '.tiff'
                if os.path.exists(fname):
                    print(f"  However, file was successfully created at: {fname}")
                    # If expected is None and result is None, this should pass
                    # If both are None but recursive_check says fail, 
                    # it might be a comparison issue
                    if expected is None and result is None:
                        print("  Both expected and result are None. Treating as PASS.")
                        print("TEST PASSED")
                        sys.exit(0)
                else:
                    print(f"  File was NOT created at: {fname}")

            sys.exit(1)

    print("TEST PASSED")
    sys.exit(0)


if __name__ == '__main__':
    main()

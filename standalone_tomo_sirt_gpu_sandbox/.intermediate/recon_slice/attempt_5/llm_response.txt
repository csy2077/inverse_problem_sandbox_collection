def _analyze_obj_(obj):
    if __main__._torch_ and isinstance(obj, __main__._torch_.Tensor):
        return {'type': 'torch.Tensor', 'shape': list(obj.shape), 'dtype': str(obj.dtype),
                'device': str(obj.device)}
    if isinstance(obj, np.ndarray):
        return {'type': 'numpy.ndarray', 'shape': list(obj.shape), 'dtype': str(obj.dtype)}
    if isinstance(obj, (list, tuple)):
        return {'type': type(obj).__name__, 'length': len(obj),
                'elements': [_analyze_obj_(item) for item in obj]}
    if hasattr(obj, '__dict__'):
        methods = []
        try:
            for m in dir(obj):
                if m.startswith('_'):
                    continue
                try:
                    attr = getattr(obj, m)
                    if callable(attr):
                        methods.append(m)
                except Exception:
                    continue
        except Exception:
            pass
        return {'type': 'CustomObject', 'class_name': obj.__class__.__name__, 'public_methods': methods,
                'attributes': list(obj.__dict__.keys())}
    try:
        val_str = str(obj)
    except:
        val_str = '<non-stringifiable>'
    return {'type': type(obj).__name__, 'value_sample': val_str}


__main__._analyze_obj_ = _analyze_obj_

def _main_recon_slice(sinogram, method, pmat, parameters=None, pixel_size=1.0, offset=0):
    if type(offset) is float:
        offset = round(offset)
    if parameters is None:
        parameters = {}
    if 'iterations' in list(parameters.keys()):
        iterations = parameters['iterations']
        opts = {key: parameters[key] for key in parameters if key != 'iterations'}
    else:
        iterations = 1
        opts = parameters
    pixel_size = float(pixel_size)
    sinogram = sinogram / pixel_size
    if offset:
        sinogram = np.roll(sinogram, -offset, axis=1)
    rec = pmat.reconstruct(method, sinogram, iterations=iterations, extraOptions=opts)
    return rec

def load_pickle_file(filepath):
    """Load pickle file using multiple strategies to handle corruption or format issues."""
    file_size = os.path.getsize(filepath)
    print(f"  File size: {file_size} bytes")

    # Read raw bytes once
    with open(filepath, 'rb') as f:
        raw_data = f.read()

    print(f"  First 20 bytes (hex): {raw_data[:20].hex()}")
    print(f"  Last 20 bytes (hex): {raw_data[-20:].hex()}")

    # Check if the file might be truncated or have extra bytes
    # Pickle protocol 2+ starts with \x80\x0N where N is protocol version
    if raw_data[0:1] == b'\x80':
        proto = raw_data[1]
        print(f"  Pickle protocol: {proto}")

    # Strategy 1: Standard dill load
    try:
        data = dill.loads(raw_data)
        print("  Loaded with dill.loads")
        return data
    except Exception as e:
        print(f"  dill.loads failed: {type(e).__name__}: {e}")

    # Strategy 2: Standard pickle load
    try:
        data = pickle.loads(raw_data)
        print("  Loaded with pickle.loads")
        return data
    except Exception as e:
        print(f"  pickle.loads failed: {type(e).__name__}: {e}")

    # Strategy 3: The file may have been written with multiple dump calls
    # or may have a corrupted trailer. Try loading just the first pickle object.
    try:
        buf = io.BytesIO(raw_data)
        unpickler = dill.Unpickler(buf)
        data = unpickler.load()
        print(f"  Loaded with dill.Unpickler (consumed {buf.tell()} of {len(raw_data)} bytes)")
        return data
    except Exception as e:
        print(f"  dill.Unpickler failed: {type(e).__name__}: {e}")

    # Strategy 4: Try pickle.Unpickler
    try:
        buf = io.BytesIO(raw_data)
        unpickler = pickle.Unpickler(buf)
        data = unpickler.load()
        print(f"  Loaded with pickle.Unpickler (consumed {buf.tell()} of {len(raw_data)} bytes)")
        return data
    except Exception as e:
        print(f"  pickle.Unpickler failed: {type(e).__name__}: {e}")

    # Strategy 5: File might have been written in append mode or double-written
    # Look for pickle STOP opcode (0x2e = '.') and try loading subsets
    stop_positions = []
    for i, b in enumerate(raw_data):
        if b == 0x2e:  # STOP opcode '.'
            stop_positions.append(i + 1)

    print(f"  Found {len(stop_positions)} potential STOP opcodes")

    # Try from smallest to largest - first valid pickle stream
    for pos in stop_positions:
        if pos < 50:  # Too small to be valid
            continue
        try:
            data = dill.loads(raw_data[:pos])
            print(f"  Loaded truncated at STOP position {pos}")
            return data
        except Exception:
            pass
        try:
            data = pickle.loads(raw_data[:pos])
            print(f"  Loaded truncated (pickle) at STOP position {pos}")
            return data
        except Exception:
            pass

    # Strategy 6: Maybe the file has been concatenated or has header bytes
    # Try skipping first few bytes
    for skip in [1, 2, 4, 8, 16]:
        try:
            data = dill.loads(raw_data[skip:])
            print(f"  Loaded by skipping first {skip} bytes")
            return data
        except Exception:
            pass

    # Strategy 7: Try with different Python pickle protocols explicitly
    for proto in [2, 3, 4, 5]:
        try:
            buf = io.BytesIO(raw_data)
            up = pickle._Unpickler(buf)
            up.encoding = 'latin1'
            data = up.load()
            print(f"  Loaded with pickle._Unpickler latin1 encoding")
            return data
        except Exception:
            pass

    # Strategy 8: Check if file contains multiple pickle streams
    # and we need the last one
    try:
        buf = io.BytesIO(raw_data)
        objects = []
        while buf.tell() < len(raw_data):
            try:
                obj = dill.load(buf)
                objects.append(obj)
            except Exception:
                break
        if objects:
            data = objects[-1]  # Return last object
            print(f"  Loaded last of {len(objects)} objects from multi-stream")
            return data
    except Exception as e:
        print(f"  Multi-stream attempt failed: {e}")

    # Strategy 9: Maybe the data file was saved with a specific dill protocol
    # or with byref/recurse settings
    for recurse_val in [True, False]:
        for byref_val in [True, False]:
            try:
                old_settings = {}
                if hasattr(dill, 'settings'):
                    old_settings = dict(dill.settings)
                    dill.settings['recurse'] = recurse_val
                    dill.settings['byref'] = byref_val
                data = dill.loads(raw_data)
                if hasattr(dill, 'settings'):
                    dill.settings.update(old_settings)
                print(f"  Loaded with dill recurse={recurse_val}, byref={byref_val}")
                return data
            except Exception:
                if hasattr(dill, 'settings') and old_settings:
                    dill.settings.update(old_settings)

    # Strategy 10: The "Ran out of input" error often means the file is incomplete
    # or the pickle references objects that need more data. Let's check if perhaps
    # there's a companion file or the data was split.
    # Also try loading with shelve-like approach
    print("  Attempting raw byte analysis...")
    
    # Check for common patterns - maybe it's a numpy .npy or .npz file disguised as .pkl
    if raw_data[:6] == b'\x93NUMPY':
        try:
            buf = io.BytesIO(raw_data)
            data = np.load(buf, allow_pickle=True)
            print("  Loaded as numpy file")
            return data
        except Exception as e:
            print(f"  numpy load failed: {e}")

    # Strategy 11: Custom unpickler that's more forgiving
    class ForgivingUnpickler(pickle.Unpickler):
        def find_class(self, module, name):
            # Handle __main__ references
            if module == '__main__':
                if hasattr(__main__, name):
                    return getattr(__main__, name)
            try:
                return super().find_class(module, name)
            except Exception:
                # Return a dummy class
                print(f"  Warning: Could not find {module}.{name}, using dummy")
                return type(name, (), {})

    try:
        buf = io.BytesIO(raw_data)
        up = ForgivingUnpickler(buf)
        up.encoding = 'latin1'
        data = up.load()
        print("  Loaded with ForgivingUnpickler (latin1)")
        return data
    except Exception as e:
        print(f"  ForgivingUnpickler latin1 failed: {type(e).__name__}: {e}")

    try:
        buf = io.BytesIO(raw_data)
        up = ForgivingUnpickler(buf)
        data = up.load()
        print("  Loaded with ForgivingUnpickler (default)")
        return data
    except Exception as e:
        print(f"  ForgivingUnpickler default failed: {type(e).__name__}: {e}")

    # Strategy 12: Maybe the file was written with protocol 0 (text) but has binary data
    try:
        with open(filepath, 'r') as f:
            text = f.read()
        data = eval(text)
        print("  Loaded as eval'd text")
        return data
    except Exception:
        pass

    # Strategy 13: file might need specific modules imported before unpickling
    # Try importing common modules that might be referenced
    modules_to_try = [
        'astra', 'scipy', 'scipy.sparse', 'skimage', 'PIL',
        'tomopy', 'tigre', 'cupy'
    ]
    for mod_name in modules_to_try:
        try:
            mod = __import__(mod_name)
            sys.modules[mod_name] = mod
            setattr(__main__, mod_name.split('.')[0], mod)
        except ImportError:
            pass

    # Retry after importing additional modules
    try:
        data = dill.loads(raw_data)
        print("  Loaded with dill after importing extra modules")
        return data
    except Exception as e:
        print(f"  dill after extra imports failed: {type(e).__name__}: {e}")

    raise RuntimeError(f"Could not load {filepath} after all attempts")

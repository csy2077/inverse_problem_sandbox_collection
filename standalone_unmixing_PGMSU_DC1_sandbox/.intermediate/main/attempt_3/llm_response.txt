import logging

# Fix logging import before agent_main is imported
# We need to patch the agent_main module's globals before it executes
# But since it executes at import time, we need to inject logging into its namespace
# The approach: modify the module file or use importlib tricks

# First, ensure logging is available globally
import builtins

# Pre-patch: create a temporary module loader
import importlib
import types

def _patch_and_import_agent_main():
    """Import agent_main with logging available in its namespace."""
    module_path = os.path.join(os.path.dirname(os.path.abspath(__file__)), 'agent_main.py')
    
    with open(module_path, 'r') as f:
        source = f.read()
    
    # Check if 'import logging' is missing
    if 'import logging' not in source:
        source = 'import logging\n' + source
    
    module = types.ModuleType('agent_main')
    module.__file__ = module_path
    module.__package__ = None
    
    # Set up the module's namespace
    module.__dict__['logging'] = logging
    module.__dict__['__name__'] = 'agent_main'
    module.__dict__['__file__'] = module_path
    
    sys.modules['agent_main'] = module
    
    compiled = compile(source, module_path, 'exec')
    exec(compiled, module.__dict__)
    
    return module

agent_main_module = _patch_and_import_agent_main()
main = agent_main_module.main
